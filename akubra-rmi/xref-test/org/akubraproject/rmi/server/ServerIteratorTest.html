48  public class ServerIteratorTest {
49    private Exporter exporter;
50  
51    @BeforeSuite
52    public void setUp() throws Exception {
53      exporter = new Exporter(0);
54    }
55  
56    @AfterSuite
57    public void tearDown() throws Exception {
58      exporter = null;
59    }
60  
61    @Test
62    public void testServerIterator() throws RemoteException {
63      ServerIterator<URI> it = new ServerIterator<URI>(new ArrayList<URI>().iterator(), exporter);
64      assertTrue(it.getExported() instanceof RemoteIterator<?>);
65    }
66  
67    @Test
68    public void testNext() throws RemoteException {
69      List<String>           l  = Arrays.asList("quick", "brown", "fox");
70      ServerIterator<String> it = new ServerIterator<String>(l.iterator(), exporter);
71  
72      for (int v : new int[] { -1, 0 }) {
73        try {
74          it.next(-1);
75          fail("Failed to recieve expected exception for batch-size: " + v);
76        } catch (Exception e) {
77        }
78      }
79  
80      List<String> n = it.next(10);
81      assertEquals(l, n);
82      n = it.next(10);
83      assertTrue(n.isEmpty());
84    }
85  
86    @Test
87    public void testUnExport() throws RemoteException {
88      List<String>           l      = Arrays.asList("quick", "brown", "fox");
89      final boolean[]        closed = new boolean[] { false };
90      ServerIterator<String> it     =
91        new ServerIterator<String>(l.iterator(), exporter) {
92          private static final long serialVersionUID = 1L;
93  
94          @Override
95          public void unExport(boolean force) {
96            closed[0] = true;
97            super.unExport(force);
98          }
99        };
100 
101     assertFalse(closed[0]);
102 
103     for (String v : l) {
104       assertEquals(v, it.next(1).get(0));
105       assertFalse(closed[0]);
106     }
107 
108     assertTrue(it.next(1).isEmpty());
109     assertTrue(closed[0]);
110   }
