144   private static FSBlob getFSBlob(String id) {
145     try {
146       URI uri = null;
147       if (id != null)
148         uri = new URI(id);
149       return new FSBlob(conn, baseDir, uri, mgr, null);
150     } catch (URISyntaxException e) {
151       throw new RuntimeException(e);
152     } catch (NullPointerException e) {
153       return null;
154     } catch (UnsupportedIdException e) {
155       return null;
156     }
157   }
158 
159   private static void assertCanonical(String origId) {
160     assertEquals(getCanonicalId(origId), origId);
161   }
162 
163   private static void assertCanonical(String origId, String canonicalId) {
164     assertEquals(getCanonicalId(origId), canonicalId);
165   }
166 
167   private static String getCanonicalId(String id) {
168     return getFSBlob(id).getCanonicalId().toString();
169   }
170 
171   private static FSBlob createFSBlob(String id) throws Exception {
172     FSBlob blob = getFSBlob(id);
173     OutputStream out = blob.openOutputStream(-1, false);
174 
175     out.write(new String("fsblob test").getBytes());
176     out.close();
177 
178     return blob;
179   }

181   private static void assertMoved(Map<String,String> hints) throws Exception {
182     FSBlob source = createFSBlob("file:source");
183     String dest = "file:dest";
184     URI dest_uri = new URI(dest);
185 
186     long size = source.getSize();
187 
188     source.moveTo(dest_uri, hints);
189 
190     assertFalse(source.exists());
191     assertTrue(getFSBlob(dest).exists());
192     assertEquals(size, getFSBlob(dest).getSize());
193 
194     getFSBlob(dest).delete();
195   }
